title: PageLoading Component
---



This component provides page loading functionality, displaying a spinner or message while content is loading.

## Props

- `prop1`: Description of prop1.
- `prop2`: Description of prop2.

## Usage

```tsx
import { PageLoading } from '@mycrumbs-uikit';

const props = {};

<PageLoading prop1="value1" prop2="value2" />;
```

```tsx
import type { FC } from 'react';

import { Spinner } from '../src/Spinner';

interface PageLoadingProps {
  message?: string;
}

export const PageLoading: FC<PageLoadingProps> = ({ message }) => {
  return (
    <div className="flex h-full grow items-center justify-center">
      <div className="space-y-3">
        <Spinner className="mx-auto" />
        <div>{message}</div>
      </div>
    </div>
  );
};

```
